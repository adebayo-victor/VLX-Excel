def create_feedback_table_cs50():
    """Creates the 'feedback' table using the CS50 SQL library."""
    try:
        # Define and execute the SQL command
        # The syntax is the same as raw SQL
        db.execute("""
            CREATE TABLE IF NOT EXISTS feedback (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                academic_year TEXT NOT NULL,
                location TEXT NOT NULL,
                problem_description TEXT NOT NULL,
                solution_suggestion TEXT,
                timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        print(f"✅ Success! Table 'feedback' created (or already exists) in {DB_FILE}")

    except Exception as e:
